 88    def _generate_session_id(self) -> str:
 89        """Generate a session ID in qa-YYYYMMDD-HHMMSS format."""
 90        now = datetime.now(timezone.utc)
 91        QAOrchestrator._session_counter += 1
 92        # Include counter to ensure uniqueness within same second
 93        base_id = f"qa-{now.strftime('%Y%m%d')}-{now.strftime('%H%M%S')}"
 94        if QAOrchestrator._session_counter > 1:
 95            return f"{base_id}-{QAOrchestrator._session_counter:03d}"
 96        return base_id

113    def test(
114        self,
115        target: str,
116        depth: QADepth = QADepth.STANDARD,
117        trigger: QATrigger = QATrigger.USER_COMMAND,
118        base_url: Optional[str] = None,
119        timeout: int = 120,
120    ) -> QASession:
121        """
122        Main entry point for QA testing.
123
124        Args:
125            target: What to test (file path, endpoint, or description).
126            depth: Testing depth level.
127            trigger: What initiated this QA session.
128            base_url: Optional base URL for API requests.
129            timeout: Timeout in seconds.
130
131        Returns:
132            QASession with test results.
133        """
134        # Generate session
135        session_id = self._generate_session_id()
136        context = QAContext(
137            target_files=[target] if target.endswith(".py") else [],
138            base_url=base_url,
139        )
140        if not target.endswith(".py") and target.startswith("/"):
141            context.target_endpoints = [QAEndpoint(method="GET", path=target)]
142
143        session = QASession(
144            session_id=session_id,
145            trigger=trigger,
146            depth=depth,
147            status=QAStatus.PENDING,
148            context=context,
149        )
150        session.started_at = datetime.now(timezone.utc).isoformat().replace("+00:00", "Z")
151
152        self._current_session = session
153        self._session_start_time = time.time()
154        self._accumulated_cost = 0.0
155        self._warnings = []
156
157        self._log("qa_session_start", {
158            "session_id": session_id,
159            "target": target,
160            "depth": depth.value,
161            "trigger": trigger.value,
162        })
163
164        try:
165            # Check cost limit before starting
166            if self._get_current_cost() >= self.limits.max_cost_usd:
167                session.status = QAStatus.COMPLETED_PARTIAL
168                session.result = QAResult(
169                    partial_completion_reason="cost_limit: exceeded max_cost_usd before starting"
170                )
171                session.completed_at = datetime.now(timezone.utc).isoformat().replace("+00:00", "Z")
172                self._save_session(session)
173                return session
174
175            # Check timeout
176            if self._check_timeout():
177                session.status = QAStatus.COMPLETED_PARTIAL
178                session.result = QAResult(
179                    partial_completion_reason="timeout: exceeded session_timeout before starting"
180                )
181                session.completed_at = datetime.now(timezone.utc).isoformat().replace("+00:00", "Z")
182                self._save_session(session)
183                return session
184
185            # Dispatch to agents
186            session.status = QAStatus.RUNNING
187            agent_results = self.dispatch_agents(depth, context)
188
189            # Aggregate results
190            result = self._aggregate_results(agent_results)
191            session.result = result
192            session.status = QAStatus.COMPLETED
193            session.completed_at = datetime.now(timezone.utc).isoformat().replace("+00:00", "Z")
194            session.cost_usd = self._accumulated_cost
195
196        except Exception as e:
197            self._log("qa_session_error", {"error": str(e)}, level="error")
198            session.status = QAStatus.BLOCKED
199            session.error = str(e)
200            session.completed_at = datetime.now(timezone.utc).isoformat().replace("+00:00", "Z")
201
202        self._save_session(session)
203        return session

304    def dispatch_agents(
305        self,
306        depth: QADepth,
307        context: QAContext,
308    ) -> dict[str, Any]:
309        """
310        Route to appropriate sub-agents based on depth.
311
312        Depth-based dispatch (Section 6):
313        - SHALLOW: BehavioralTester only (happy path)
314        - STANDARD: BehavioralTester + ContractValidator
315        - DEEP: All three agents + security probes
316        - REGRESSION: RegressionScanner + targeted BehavioralTester
317
318        Args:
319            depth: Testing depth level.
320            context: QAContext for the session.
321
322        Returns:
323            Dictionary with results from each agent.
324        """
325        results: dict[str, Any] = {}
326        skipped_reasons: dict[str, str] = {}
327
328        # Build agent context
329        agent_context = {
330            "base_url": context.base_url or "http://localhost:8000",
331            "endpoints": context.target_endpoints,
332            "target_files": context.target_files,
333            "depth": depth,
334            "skip_service_start": True,  # Default to not auto-starting
335        }
336
337        # Limit endpoints based on depth
338        if context.target_endpoints:
339            agent_context["endpoints"] = self._limit_endpoints(
340                context.target_endpoints, depth
341            )
342
343        if depth == QADepth.SHALLOW:
344            # SHALLOW: BehavioralTester only
345            results["behavioral"] = self._run_agent_safe(
346                self.behavioral_agent, agent_context, "behavioral", skipped_reasons
347            )
348
349        elif depth == QADepth.STANDARD:
350            # STANDARD: BehavioralTester + ContractValidator
351            results["behavioral"] = self._run_agent_safe(
352                self.behavioral_agent, agent_context, "behavioral", skipped_reasons
353            )
354            results["contract"] = self._run_agent_safe(
355                self.contract_agent, agent_context, "contract", skipped_reasons
356            )
357
358        elif depth == QADepth.DEEP:
359            # DEEP: All three agents
360            results["behavioral"] = self._run_agent_safe(
361                self.behavioral_agent, agent_context, "behavioral", skipped_reasons
362            )
363            results["contract"] = self._run_agent_safe(
364                self.contract_agent, agent_context, "contract", skipped_reasons
365            )
366            results["regression"] = self._run_agent_safe(
367                self.regression_agent, agent_context, "regression", skipped_reasons
368            )
369
370        elif depth == QADepth.REGRESSION:
371            # REGRESSION: RegressionScanner + targeted BehavioralTester
372            regression_result = self._run_agent_safe(
373                self.regression_agent, agent_context, "regression", skipped_reasons
374            )
375            results["regression"] = regression_result
376
377            # Use regression scanner's output to target behavioral tests
378            if regression_result and isinstance(regression_result, dict):
379                regression_suite = regression_result.get("regression_suite", {})
380                must_test = regression_suite.get("must_test", [])
381                should_test = regression_suite.get("should_test", [])
382                targeted_endpoints = [
383                    QAEndpoint(method="GET", path=path)
384                    for path in must_test + should_test
385                ]
386                if targeted_endpoints:
387                    agent_context["endpoints"] = targeted_endpoints
388
389            results["behavioral"] = self._run_agent_safe(
390                self.behavioral_agent, agent_context, "behavioral", skipped_reasons
391            )
392
393        if skipped_reasons:
394            results["skipped_reasons"] = skipped_reasons
395
396        return results

398    def _run_agent_safe(
399        self,
400        agent: Any,
401        context: dict[str, Any],
402        agent_name: str,
403        skipped_reasons: dict[str, str],
404    ) -> Optional[dict[str, Any]]:
405        """
406        Run an agent safely, catching exceptions.
407
408        Args:
409            agent: The agent to run.
410            context: Context for the agent.
411            agent_name: Name of the agent for logging.
412            skipped_reasons: Dict to track skipped agents.
413
414        Returns:
415            Agent output or None if failed.
416        """
417        try:
418            result = agent.run(context)
419            if result.success:
420                return result.output
421            else:
422                skipped_reasons[agent_name] = result.error or "Agent failed"
423                return None
424        except Exception as e:
425            self._log(f"{agent_name}_agent_error", {"error": str(e)}, level="error")
426            skipped_reasons[agent_name] = str(e)
427            return None

429    def _aggregate_results(self, agent_results: dict[str, Any]) -> QAResult:
430        """
431        Combine results from all agents.
432
433        Args:
434            agent_results: Dictionary of results from each agent.
435
436        Returns:
437            Aggregated QAResult.
438        """
439        result = QAResult()
440        all_findings: list[QAFinding] = []
441
442        # Extract behavioral results
443        if behavioral := agent_results.get("behavioral"):
444            if isinstance(behavioral, dict):
445                result.tests_run += behavioral.get("tests_run", 0)
446                result.tests_passed += behavioral.get("tests_passed", 0)
447                result.tests_failed += behavioral.get("tests_failed", 0)
448                findings = behavioral.get("findings", [])
449                for f in findings:
450                    if isinstance(f, QAFinding):
451                        all_findings.append(f)
452                result.behavioral_results = behavioral
453
454        # Extract contract results
455        if contract := agent_results.get("contract"):
456            if isinstance(contract, dict):
457                contracts_checked = contract.get("contracts_checked", 0)
458                result.tests_run += contracts_checked
459                result.tests_passed += contract.get("contracts_valid", 0)
460                result.tests_failed += contract.get("contracts_broken", 0)
461                findings = contract.get("findings", [])
462                for f in findings:
463                    if isinstance(f, QAFinding):
464                        all_findings.append(f)
465                result.contract_results = contract
466
467        # Extract regression results
468        if regression := agent_results.get("regression"):
469            if isinstance(regression, dict):
470                result.regression_results = regression
471                findings = regression.get("findings", [])
472                for f in findings:
473                    if isinstance(f, QAFinding):
474                        all_findings.append(f)
475
476        # Track skipped reasons
477        if skipped := agent_results.get("skipped_reasons"):
478            result.skipped_reasons = skipped
479
480        # Deduplicate findings (by endpoint + actual status)
481        seen_keys: set[tuple] = set()
482        unique_findings: list[QAFinding] = []
483        for finding in all_findings:
484            key = (
485                finding.endpoint,
486                str(finding.actual.get("status", "")),
487                finding.title,
488            )
489            if key not in seen_keys:
490                seen_keys.add(key)
491                unique_findings.append(finding)
492
493        result.findings = unique_findings
494
495        # Count by severity
496        for finding in result.findings:
497            if finding.severity == "critical":
498                result.critical_count += 1
499            elif finding.severity == "moderate":
500                result.moderate_count += 1
501            else:
502                result.minor_count += 1
503
504        # Set recommendation based on findings
505        if result.critical_count > 0:
506            result.recommendation = QARecommendation.BLOCK
507        elif result.moderate_count > 0:
508            result.recommendation = QARecommendation.WARN
509        else:
510            result.recommendation = QARecommendation.PASS
511
512        return result

562    def _save_report(self, session: QASession) -> None:
563        """Generate and save QA report."""
564        session_dir = self.sessions_path / session.session_id
565        report_path = session_dir / "qa-report.md"
566
567        result = session.result
568        if not result:
569            report_path.write_text(f"# QA Report: {session.session_id}\n\nNo results available.")
570            return
571
572        lines = [
573            f"# QA Report: {session.session_id}",
574            "",
575            f"**Trigger:** {session.trigger.value}",
576            f"**Depth:** {session.depth.value}",
577            f"**Status:** {session.status.value}",
578            f"**Recommendation:** {result.recommendation.value.upper()}",
579            "",
580            "## Summary",
581            "",
582            f"- **Tests Run:** {result.tests_run}",
583            f"- **Passed:** {result.tests_passed}",
584            f"- **Failed:** {result.tests_failed}",
585            f"- **Endpoints Tested:** {len(result.endpoints_tested)}",
586            "",
587            "## Findings",
588            "",
589        ]
590
591        if not result.findings:
592            lines.append("No issues found.")
593        else:
594            for finding in result.findings:
595                lines.extend([
596                    f"### [{finding.severity.upper()}] {finding.finding_id}: {finding.title}",
597                    "",
598                    f"**Endpoint:** {finding.endpoint}",
599                    f"**Category:** {finding.category}",
600                    "",
601                    finding.description,
602                    "",
603                    "**Expected:**",
604                    "```json",
605                    json.dumps(finding.expected, indent=2),
606                    "```",
607                    "",
608                    "**Actual:**",
609                    "```json",
610                    json.dumps(finding.actual, indent=2),
611                    "```",
612                    "",
613                    f"**Recommendation:** {finding.recommendation}",
614                    "",
615                    "---",
616                    "",
617                ])
618
619        report_path.write_text("\n".join(lines))

621    def get_session(self, session_id: str) -> Optional[QASession]:
622        """

625        Args:

631        state_file = self.sessions_path / session_id / "state.json"

642    def list_sessions(self, limit: int = 20) -> list[str]:
643        """
644        List recent QA session IDs.
645
646        Args:
647            limit: Maximum number of sessions to return.
648
649        Returns:
650            List of session IDs, newest first.
651        """
652        sessions = []

655

659                if len(sessions) >= limit:

664    def get_findings(
665        self,
666        session_id: Optional[str] = None,
667        severity: Optional[str] = None,
668    ) -> list[QAFinding]:
669        """
670        Get findings, optionally filtered.
671
672        Args:
673            session_id: Optional session ID to filter by.
674            severity: Optional severity to filter by.
675
676        Returns:
677            List of QAFinding objects.
678        """
679        findings: list[QAFinding] = []
680
681        if session_id:
682            # Get findings from specific session
683            session = self.get_session(session_id)
684            if session and session.result:
685                findings = session.result.findings
686        else:
687            # Get findings from all recent sessions
688            for sid in self.list_sessions(limit=50):
689                session = self.get_session(sid)
690                if session and session.result:
691                    findings.extend(session.result.findings)
692
693        # Filter by severity if specified
694        if severity:
695            findings = [f for f in findings if f.severity == severity]
696
697        return findings

305    def dispatch_agents(
306        self,
307        depth: QADepth,
308        context: QAContext,
309    ) -> dict[str, Any]:
310        """
311        Route to appropriate sub-agents based on depth.
312
313        Depth-based dispatch (Section 6):
314        - SHALLOW: BehavioralTester only (happy path)
315        - STANDARD: BehavioralTester + ContractValidator
316        - DEEP: All three agents + security probes
317        - REGRESSION: RegressionScanner + targeted BehavioralTester
318
319        Args:
320            depth: Testing depth level.
321            context: QAContext for the session.
322
323        Returns:
324            Dictionary with results from each agent.
325        """
326        results: dict[str, Any] = {}
327        skipped_reasons: dict[str, str] = {}
328
329        # Build agent context
330        agent_context = {
331            "base_url": context.base_url or "http://localhost:8000",
332            "endpoints": context.target_endpoints,
333            "target_files": context.target_files,
334            "depth": depth,
335            "skip_service_start": True,  # Default to not auto-starting
336        }
337
338        # Limit endpoints based on depth
339        if context.target_endpoints:
340            agent_context["endpoints"] = self._limit_endpoints(
341                context.target_endpoints, depth
342            )
343
344        if depth == QADepth.SHALLOW:
345            # SHALLOW: BehavioralTester only
346            results["behavioral"] = self._run_agent_safe(
347                self.behavioral_agent, agent_context, "behavioral", skipped_reasons
348            )
349
350        elif depth == QADepth.STANDARD:
351            # STANDARD: BehavioralTester + ContractValidator
352            results["behavioral"] = self._run_agent_safe(
353                self.behavioral_agent, agent_context, "behavioral", skipped_reasons
354            )
355            results["contract"] = self._run_agent_safe(
356                self.contract_agent, agent_context, "contract", skipped_reasons
357            )
358
359        elif depth == QADepth.DEEP:
360            # DEEP: All three agents
361            results["behavioral"] = self._run_agent_safe(
362                self.behavioral_agent, agent_context, "behavioral", skipped_reasons
363            )
364            results["contract"] = self._run_agent_safe(
365                self.contract_agent, agent_context, "contract", skipped_reasons
366            )
367            results["regression"] = self._run_agent_safe(
368                self.regression_agent, agent_context, "regression", skipped_reasons
369            )
370
371        elif depth == QADepth.REGRESSION:
372            # REGRESSION: RegressionScanner + targeted BehavioralTester
373            regression_result = self._run_agent_safe(
374                self.regression_agent, agent_context, "regression", skipped_reasons
375            )
376            results["regression"] = regression_result
377
378            # Use regression scanner's output to target behavioral tests
379            if regression_result and isinstance(regression_result, dict):
380                regression_suite = regression_result.get("regression_suite", {})
381                must_test = regression_suite.get("must_test", [])
382                should_test = regression_suite.get("should_test", [])
383                targeted_endpoints = [
384                    QAEndpoint(method="GET", path=path)
385                    for path in must_test + should_test
386                ]
387                if targeted_endpoints:
388                    agent_context["endpoints"] = targeted_endpoints
389
390            results["behavioral"] = self._run_agent_safe(
391                self.behavioral_agent, agent_context, "behavioral", skipped_reasons
392            )
393
394        if skipped_reasons:
395            results["skipped_reasons"] = skipped_reasons
396
397        return results

400        self,
401        agent: Any,
402        context: dict[str, Any],
403        agent_name: str,
404        skipped_reasons: dict[str, str],
405    ) -> Optional[dict[str, Any]]:
406        """
407        Run an agent safely, catching exceptions.
408
409        Args:
410            agent: The agent to run.
411            context: Context for the agent.

413            skipped_reasons: Dict to track skipped agents.

416            Agent output or None if failed.
417        """

421                return result.output
422            else:

424                return None
425        except Exception as e:

427            skipped_reasons[agent_name] = str(e)

430    def _aggregate_results(self, agent_results: dict[str, Any]) -> QAResult:
431        """
432        Combine results from all agents.
433
434        Args:
435            agent_results: Dictionary of results from each agent.
436
437        Returns:
438            Aggregated QAResult.
439        """
440        result = QAResult()
441        all_findings: list[QAFinding] = []
442
443        # Extract behavioral results
444        if behavioral := agent_results.get("behavioral"):
445            if isinstance(behavioral, dict):
446                result.tests_run += behavioral.get("tests_run", 0)
447                result.tests_passed += behavioral.get("tests_passed", 0)
448                result.tests_failed += behavioral.get("tests_failed", 0)
449                findings = behavioral.get("findings", [])
450                for f in findings:
451                    if isinstance(f, QAFinding):
452                        all_findings.append(f)
453                result.behavioral_results = behavioral
454
455        # Extract contract results
456        if contract := agent_results.get("contract"):
457            if isinstance(contract, dict):
458                contracts_checked = contract.get("contracts_checked", 0)
459                result.tests_run += contracts_checked
460                result.tests_passed += contract.get("contracts_valid", 0)
461                result.tests_failed += contract.get("contracts_broken", 0)
462                findings = contract.get("findings", [])
463                for f in findings:
464                    if isinstance(f, QAFinding):
465                        all_findings.append(f)
466                result.contract_results = contract
467
468        # Extract regression results
469        if regression := agent_results.get("regression"):
470            if isinstance(regression, dict):
471                result.regression_results = regression
472                findings = regression.get("findings", [])
473                for f in findings:
474                    if isinstance(f, QAFinding):
475                        all_findings.append(f)
476
477        # Track skipped reasons
478        if skipped := agent_results.get("skipped_reasons"):
479            result.skipped_reasons = skipped
480
481        # Deduplicate findings (by endpoint + actual status)
482        seen_keys: set[tuple] = set()
483        unique_findings: list[QAFinding] = []
484        for finding in all_findings:
485            key = (
486                finding.endpoint,
487                str(finding.actual.get("status", "")),
488                finding.title,
489            )
490            if key not in seen_keys:
491                seen_keys.add(key)
492                unique_findings.append(finding)
493
494        result.findings = unique_findings
495
496        # Count by severity
497        for finding in result.findings:
498            if finding.severity == "critical":
499                result.critical_count += 1
500            elif finding.severity == "moderate":
501                result.moderate_count += 1
502            else:
503                result.minor_count += 1
504
505        # Set recommendation based on findings
506        if result.critical_count > 0:
507            result.recommendation = QARecommendation.BLOCK
508        elif result.moderate_count > 0:
509            result.recommendation = QARecommendation.WARN
510        else:
511            result.recommendation = QARecommendation.PASS
512
513        return result

305    def dispatch_agents(
306        self,
307        depth: QADepth,
308        context: QAContext,
309    ) -> dict[str, Any]:
310        """
311        Route to appropriate sub-agents based on depth.
312
313        Depth-based dispatch (Section 6):
314        - SHALLOW: BehavioralTester only (happy path)
315        - STANDARD: BehavioralTester + ContractValidator
316        - DEEP: All three agents + security probes
317        - REGRESSION: RegressionScanner + targeted BehavioralTester
318
319        Args:
320            depth: Testing depth level.
321            context: QAContext for the session.
322
323        Returns:
324            Dictionary with results from each agent.
325        """
326        results: dict[str, Any] = {}
327        skipped_reasons: dict[str, str] = {}
328
329        # Build agent context
330        agent_context = {
331            "base_url": context.base_url or "http://localhost:8000",
332            "endpoints": context.target_endpoints,
333            "target_files": context.target_files,
334            "depth": depth,
335            "skip_service_start": True,  # Default to not auto-starting
336        }
337
338        # Limit endpoints based on depth
339        if context.target_endpoints:
340            agent_context["endpoints"] = self._limit_endpoints(
341                context.target_endpoints, depth
342            )
343
344        if depth == QADepth.SHALLOW:
345            # SHALLOW: BehavioralTester only
346            results["behavioral"] = self._run_agent_safe(
347                self.behavioral_agent, agent_context, "behavioral", skipped_reasons
348            )
349
350        elif depth == QADepth.STANDARD:
351            # STANDARD: BehavioralTester + ContractValidator
352            results["behavioral"] = self._run_agent_safe(
353                self.behavioral_agent, agent_context, "behavioral", skipped_reasons
354            )
355            results["contract"] = self._run_agent_safe(
356                self.contract_agent, agent_context, "contract", skipped_reasons
357            )
358
359        elif depth == QADepth.DEEP:
360            # DEEP: All three agents
361            results["behavioral"] = self._run_agent_safe(
362                self.behavioral_agent, agent_context, "behavioral", skipped_reasons
363            )
364            results["contract"] = self._run_agent_safe(
365                self.contract_agent, agent_context, "contract", skipped_reasons
366            )
367            results["regression"] = self._run_agent_safe(
368                self.regression_agent, agent_context, "regression", skipped_reasons
369            )
370
371        elif depth == QADepth.REGRESSION:
372            # REGRESSION: RegressionScanner + targeted BehavioralTester
373            regression_result = self._run_agent_safe(
374                self.regression_agent, agent_context, "regression", skipped_reasons
375            )
376            results["regression"] = regression_result
377
378            # Use regression scanner's output to target behavioral tests
379            if regression_result and isinstance(regression_result, dict):
380                regression_suite = regression_result.get("regression_suite", {})
381                must_test = regression_suite.get("must_test", [])
382                should_test = regression_suite.get("should_test", [])
383                targeted_endpoints = [
384                    QAEndpoint(method="GET", path=path)
385                    for path in must_test + should_test
386                ]
387                if targeted_endpoints:
388                    agent_context["endpoints"] = targeted_endpoints
389
390            results["behavioral"] = self._run_agent_safe(
391                self.behavioral_agent, agent_context, "behavioral", skipped_reasons
392            )
393
394        if skipped_reasons:
395            results["skipped_reasons"] = skipped_reasons
396
397        return results
