 72    def build_context(
 73        self,
 74        trigger: QATrigger,
 75        target: str,
 76        feature_id: Optional[str] = None,
 77        issue_number: Optional[int] = None,
 78        bug_id: Optional[str] = None,
 79        base_url: Optional[str] = None,
 80        explicit_endpoints: Optional[list[QAEndpoint]] = None,
 81    ) -> QAContext:
 82        """
 83        Build complete QA context from available sources.
 84
 85        Args:
 86            trigger: What initiated this QA session.
 87            target: What to test (file path, endpoint, or description).
 88            feature_id: Optional feature identifier.
 89            issue_number: Optional issue number.
 90            bug_id: Optional bug identifier.
 91            base_url: Optional base URL for API requests.
 92            explicit_endpoints: Optional list of explicitly specified endpoints.
 93
 94        Returns:
 95            QAContext with all gathered information.
 96        """
 97        self._log("build_context_start", {
 98            "trigger": trigger.value,
 99            "target": target,
100            "feature_id": feature_id,
101        })
102
103        context = QAContext(
104            feature_id=feature_id,
105            issue_number=issue_number,
106            bug_id=bug_id,
107            base_url=base_url,
108        )
109
110        # Set target files
111        target_path = Path(target)
112        if target_path.is_absolute() and target_path.exists():
113            context.target_files = [str(target_path)]
114        elif (self.repo_root / target).exists():
115            context.target_files = [str(self.repo_root / target)]
116
117        # Use explicit endpoints if provided, otherwise discover
118        if explicit_endpoints:
119            context.target_endpoints = explicit_endpoints
120        else:
121            # Discover endpoints from target
122            if context.target_files:
123                context.target_endpoints = self.discover_endpoints(context.target_files[0])
124            elif target.startswith("/"):
125                # Target is an endpoint path
126                context.target_endpoints = [QAEndpoint(method="GET", path=target)]
127
128        # Load spec content for post-verification triggers
129        if feature_id:
130            context.spec_content = self._load_spec_content(feature_id)
131            if issue_number:
132                issue_content = self._load_issue_content(feature_id, issue_number)
133                if issue_content:
134                    if context.spec_content:
135                        context.spec_content += f"\n\n## Issue #{issue_number}\n{issue_content}"
136                    else:
137                        context.spec_content = issue_content
138
139        # Get git diff for pre-merge/regression triggers
140        if trigger == QATrigger.PRE_MERGE:
141            context.git_diff = self._get_git_diff()
142
143        # Find related test files
144        if context.target_files:
145            context.related_tests = self._find_related_tests(context.target_files[0])
146
147        self._log("build_context_complete", {
148            "endpoints_found": len(context.target_endpoints),
149            "files_found": len(context.target_files),
150        })
151
152        return context

154    def discover_endpoints(self, target: str) -> list[QAEndpoint]:
155        """
156        Discover API endpoints from code/specs.
157
158        Searches for endpoints in:
159        1. OpenAPI/Swagger specs
160        2. FastAPI/Flask route decorators
161        3. Other common patterns
162
163        Args:
164            target: File path or description to search.
165
166        Returns:
167            List of discovered QAEndpoint objects.
168        """
169        endpoints: list[QAEndpoint] = []
170
171        target_path = Path(target)
172        if not target_path.exists():
173            # Try relative to repo root
174            target_path = self.repo_root / target
175            if not target_path.exists():
176                return endpoints
177
178        # Check if it's an OpenAPI spec file
179        if target_path.suffix in [".yaml", ".yml", ".json"]:
180            endpoints.extend(self._discover_from_openapi(target_path))
181        elif target_path.suffix == ".py":
182            endpoints.extend(self._discover_from_python(target_path))
183        elif target_path.suffix in [".ts", ".js"]:
184            endpoints.extend(self._discover_from_typescript(target_path))
185
186        return endpoints

211    def _discover_from_openapi(self, spec_path: Path) -> list[QAEndpoint]:
212        """Discover endpoints from OpenAPI/Swagger spec."""
213        endpoints: list[QAEndpoint] = []
214
215        try:
216            content = spec_path.read_text()
217
218            if spec_path.suffix == ".json":
219                spec = json.loads(content)
220            else:
221                # Simple YAML parsing for paths
222                spec = self._parse_simple_yaml(content)
223
224            paths = spec.get("paths", {})
225            for path, methods in paths.items():
226                if isinstance(methods, dict):
227                    for method in methods.keys():
228                        if method.upper() in ["GET", "POST", "PUT", "DELETE", "PATCH"]:
229                            endpoints.append(QAEndpoint(
230                                method=method.upper(),
231                                path=path,
232                                auth_required=self._check_auth_in_openapi(methods.get(method, {})),
233                            ))
234        except Exception as e:
235            self._log("openapi_parse_error", {"error": str(e)}, level="warning")
236
237        return endpoints

239    def _parse_simple_yaml(self, content: str) -> dict[str, Any]:
240        """Simple YAML parser for OpenAPI specs."""
241        result: dict[str, Any] = {"paths": {}}
242
243        current_path = None
244        lines = content.split('\n')
245
246        for i, line in enumerate(lines):
247            # Look for paths section
248            if line.strip() == "paths:":
249                # Parse paths
250                indent_level = len(line) - len(line.lstrip())
251                j = i + 1
252                while j < len(lines):
253                    path_line = lines[j]
254                    if not path_line.strip() or path_line.strip().startswith('#'):
255                        j += 1
256                        continue
257
258                    current_indent = len(path_line) - len(path_line.lstrip())
259                    if current_indent <= indent_level and path_line.strip():
260                        break
261
262                    # Check if it's a path definition (starts with /)
263                    stripped = path_line.strip()
264                    if stripped.startswith('/') or stripped.startswith("'/") or stripped.startswith('"/'):
265                        # Extract path
266                        path_match = re.match(r"['\"]?(/[^'\":]+)['\"]?:", stripped)
267                        if path_match:
268                            current_path = path_match.group(1)
269                            result["paths"][current_path] = {}
270                    elif current_path and stripped.rstrip(':') in ["get", "post", "put", "delete", "patch"]:
271                        method = stripped.rstrip(':')
272                        result["paths"][current_path][method] = {}
273
274                    j += 1
275
276        return result

284    def _discover_from_python(self, file_path: Path) -> list[QAEndpoint]:
285        """Discover endpoints from Python file (FastAPI, Flask, etc.)."""
286        endpoints: list[QAEndpoint] = []
287
288        try:
289            content = file_path.read_text()
290
291            # FastAPI patterns
292            # @router.get("/path") or @app.get("/path")
293            fastapi_patterns = [
294                (r'@(?:router|app)\.get\s*\(\s*["\']([^"\']+)["\']', "GET"),
295                (r'@(?:router|app)\.post\s*\(\s*["\']([^"\']+)["\']', "POST"),
296                (r'@(?:router|app)\.put\s*\(\s*["\']([^"\']+)["\']', "PUT"),
297                (r'@(?:router|app)\.delete\s*\(\s*["\']([^"\']+)["\']', "DELETE"),
298                (r'@(?:router|app)\.patch\s*\(\s*["\']([^"\']+)["\']', "PATCH"),
299            ]
300
301            for pattern, method in fastapi_patterns:
302                for match in re.finditer(pattern, content):
303                    path = match.group(1)
304                    # Check if auth is required (look for Depends in the line)
305                    line_start = content.rfind('\n', 0, match.start()) + 1
306                    line_end = content.find('\n', match.end())
307                    line = content[line_start:line_end if line_end > 0 else len(content)]
308                    auth_required = "Depends" in line or "get_current_user" in line
309
310                    endpoints.append(QAEndpoint(
311                        method=method,
312                        path=path,
313                        auth_required=auth_required,
314                    ))
315
316            # Flask patterns
317            # @app.route("/path", methods=["GET"])
318            flask_pattern = r'@(?:app|blueprint)\.route\s*\(\s*["\']([^"\']+)["\'](?:.*?methods\s*=\s*\[([^\]]+)\])?'
319            for match in re.finditer(flask_pattern, content, re.DOTALL):
320                path = match.group(1)
321                methods_str = match.group(2)
322
323                if methods_str:
324                    methods = re.findall(r'["\'](\w+)["\']', methods_str)
325                else:
326                    methods = ["GET"]
327
328                for method in methods:
329                    endpoints.append(QAEndpoint(
330                        method=method.upper(),
331                        path=path,
332                    ))
333
334        except Exception as e:
335            self._log("python_parse_error", {"error": str(e)}, level="warning")
336
337        return endpoints

339    def _discover_from_typescript(self, file_path: Path) -> list[QAEndpoint]:
340        """Discover endpoints from TypeScript/JavaScript file."""
341        endpoints: list[QAEndpoint] = []
342
343        try:
344            content = file_path.read_text()
345
346            # Express patterns
347            # app.get('/path', handler)
348            express_patterns = [
349                (r'\.get\s*\(\s*["\']([^"\']+)["\']', "GET"),
350                (r'\.post\s*\(\s*["\']([^"\']+)["\']', "POST"),
351                (r'\.put\s*\(\s*["\']([^"\']+)["\']', "PUT"),
352                (r'\.delete\s*\(\s*["\']([^"\']+)["\']', "DELETE"),
353            ]
354
355            for pattern, method in express_patterns:
356                for match in re.finditer(pattern, content):
357                    path = match.group(1)
358                    if path.startswith('/'):
359                        endpoints.append(QAEndpoint(
360                            method=method,
361                            path=path,
362                        ))
363
364        except Exception as e:
365            self._log("typescript_parse_error", {"error": str(e)}, level="warning")
366
367        return endpoints

369    def extract_schemas(self, endpoints: list[QAEndpoint]) -> dict[str, Any]:
370        """
371        Extract response schemas for endpoints.
372
373        Searches for schemas in:
374        1. OpenAPI spec components
375        2. Pydantic model definitions
376        3. TypedDict definitions
377
378        Args:
379            endpoints: List of endpoints to find schemas for.
380
381        Returns:
382            Dictionary mapping endpoint paths to their schemas.
383        """
384        schemas: dict[str, Any] = {}
385
386        # Search for OpenAPI spec
387        openapi_paths = list(self.repo_root.glob("**/openapi.yaml")) + \
388                       list(self.repo_root.glob("**/openapi.json")) + \
389                       list(self.repo_root.glob("**/swagger.yaml")) + \
390                       list(self.repo_root.glob("**/swagger.json"))
391
392        for spec_path in openapi_paths[:1]:  # Use first found
393            try:
394                content = spec_path.read_text()
395                if spec_path.suffix == ".json":
396                    spec = json.loads(content)
397                else:
398                    spec = self._parse_simple_yaml(content)
399
400                # Extract schemas from components
401                if "components" in spec and "schemas" in spec.get("components", {}):
402                    schemas["_components"] = spec["components"]["schemas"]
403
404            except Exception:
405                pass
406
407        return schemas

409    def find_consumers(self, endpoints: list[QAEndpoint]) -> dict[str, list[str]]:
410        """
411        Find code that calls each endpoint.
412
413        Searches for:
414        1. Frontend fetch/axios calls
415        2. Other service HTTP calls
416        3. Integration tests
417
418        Args:
419            endpoints: List of endpoints to find consumers for.
420
421        Returns:
422            Dictionary mapping endpoint paths to list of consumer file paths.
423        """
424        consumers: dict[str, list[str]] = {}
425
426        for endpoint in endpoints:
427            consumers[endpoint.path] = []
428
429        # Search patterns
430        search_patterns = [
431            # JavaScript/TypeScript fetch
432            r'fetch\s*\(\s*["\'][^"\']*({path})',
433            # Axios
434            r'axios\.(get|post|put|delete)\s*\(\s*["\'][^"\']*({path})',
435            # Python requests
436            r'requests\.(get|post|put|delete)\s*\([^)]*({path})',
437            # httpx
438            r'client\.(get|post|put|delete)\s*\([^)]*({path})',
439        ]
440
441        # Search in common directories
442        search_dirs = [
443            self.repo_root / "frontend",
444            self.repo_root / "src",
445            self.repo_root / "clients",
446            self.repo_root / "tests",
447        ]
448
449        for search_dir in search_dirs:
450            if not search_dir.exists():
451                continue
452
453            for ext in ["*.ts", "*.tsx", "*.js", "*.jsx", "*.py"]:
454                for file_path in search_dir.rglob(ext):
455                    try:
456                        content = file_path.read_text()
457
458                        for endpoint in endpoints:
459                            # Escape special regex chars in path
460                            escaped_path = re.escape(endpoint.path)
461                            # Allow for path parameters
462                            pattern_path = escaped_path.replace(r'\{', r'[^/]*').replace(r'\}', '')
463
464                            for pattern in search_patterns:
465                                full_pattern = pattern.format(path=pattern_path)
466                                if re.search(full_pattern, content):
467                                    consumers[endpoint.path].append(str(file_path))
468                                    break
469
470                            # Also check for literal path in file
471                            if endpoint.path in content:
472                                if str(file_path) not in consumers[endpoint.path]:
473                                    consumers[endpoint.path].append(str(file_path))
474
475                    except Exception:
476                        pass
477
478        return consumers

480    def _get_git_diff(self) -> Optional[str]:
481        """
482        Get git diff for current changes.
483
484        Returns:
485            Git diff as string, or None if not in git repo.
486        """
487        try:
488            result = subprocess.run(
489                ["git", "diff", "HEAD"],
490                cwd=str(self.repo_root),
491                capture_output=True,
492                text=True,
493                timeout=30,
494            )
495
496            if result.returncode != 0:
497                # Try just uncommitted changes
498                result = subprocess.run(
499                    ["git", "diff"],
500                    cwd=str(self.repo_root),
501                    capture_output=True,
502                    text=True,
503                    timeout=30,
504                )
505
506            if result.returncode != 0:
507                return None
508
509            return result.stdout if result.stdout else None
510
511        except (subprocess.TimeoutExpired, FileNotFoundError):
512            return None
513        except Exception:
514            return None

516    def _load_spec_content(self, feature_id: str) -> Optional[str]:
517        """
518        Load spec content for a feature.
519
520        Tries in order:
521        1. specs/{feature_id}/spec-final.md
522        2. specs/{feature_id}/spec-draft.md
523
524        Args:
525            feature_id: Feature identifier.
526
527        Returns:
528            Spec content as string, or None if not found.
529        """
530        spec_dir = self.repo_root / "specs" / feature_id
531
532        # Try spec-final first
533        spec_final = spec_dir / "spec-final.md"
534        if spec_final.exists():
535            return spec_final.read_text()
536
537        # Fall back to spec-draft
538        spec_draft = spec_dir / "spec-draft.md"
539        if spec_draft.exists():
540            return spec_draft.read_text()
541
542        return None

544    def _load_issue_content(
545        self, feature_id: str, issue_number: int
546    ) -> Optional[str]:
547        """
548        Load issue content from issues.json.
549
550        Args:
551            feature_id: Feature identifier.
552            issue_number: Issue number to find.
553
554        Returns:
555            Issue content as string, or None if not found.
556        """
557        issues_file = self.repo_root / "specs" / feature_id / "issues.json"
558
559        if not issues_file.exists():
560            return None
561
562        try:
563            issues = json.loads(issues_file.read_text())
564
565            for issue in issues:
566                if issue.get("number") == issue_number:
567                    title = issue.get("title", "")
568                    body = issue.get("body", "")
569                    return f"**{title}**\n\n{body}"
570
571            return None
572
573        except Exception:
574            return None

576    def _find_related_tests(self, source_file: str) -> list[str]:
577        """
578        Find test files related to source file.
579
580        Args:
581            source_file: Path to source file.
582
583        Returns:
584            List of related test file paths.
585        """
586        related: list[str] = []
587        source_path = Path(source_file)
588
589        if not source_path.exists():
590            return related
591
592        # Get the base name without extension
593        base_name = source_path.stem
594
595        # Search for test files
596        test_patterns = [
597            f"**/test_{base_name}.py",
598            f"**/{base_name}_test.py",
599            f"**/tests/**/test_{base_name}.py",
600        ]
601
602        for pattern in test_patterns:
603            for test_file in self.repo_root.glob(pattern):
604                related.append(str(test_file))
605
606        return related

212    def _discover_from_openapi(self, spec_path: Path) -> list[QAEndpoint]:
213        """Discover endpoints from OpenAPI/Swagger spec."""
214        endpoints: list[QAEndpoint] = []
215
216        try:
217            content = spec_path.read_text()
218
219            if spec_path.suffix == ".json":
220                spec = json.loads(content)
221            else:
222                # Simple YAML parsing for paths
223                spec = self._parse_simple_yaml(content)
224
225            paths = spec.get("paths", {})
226            for path, methods in paths.items():
227                if isinstance(methods, dict):
228                    for method in methods.keys():
229                        if method.upper() in ["GET", "POST", "PUT", "DELETE", "PATCH"]:
230                            endpoints.append(QAEndpoint(
231                                method=method.upper(),
232                                path=path,
233                                auth_required=self._check_auth_in_openapi(methods.get(method, {})),
234                            ))
235        except Exception as e:
236            self._log("openapi_parse_error", {"error": str(e)}, level="warning")
237
238        return endpoints

240    def _parse_simple_yaml(self, content: str) -> dict[str, Any]:
241        """Simple YAML parser for OpenAPI specs."""
242        result: dict[str, Any] = {"paths": {}}
243
244        current_path = None
245        lines = content.split('\n')
246
247        for i, line in enumerate(lines):
248            # Look for paths section
249            if line.strip() == "paths:":
250                # Parse paths
251                indent_level = len(line) - len(line.lstrip())
252                j = i + 1
253                while j < len(lines):
254                    path_line = lines[j]
255                    if not path_line.strip() or path_line.strip().startswith('#'):
256                        j += 1
257                        continue
258
259                    current_indent = len(path_line) - len(path_line.lstrip())
260                    if current_indent <= indent_level and path_line.strip():
261                        break
262
263                    # Check if it's a path definition (starts with /)
264                    stripped = path_line.strip()
265                    if stripped.startswith('/') or stripped.startswith("'/") or stripped.startswith('"/'):
266                        # Extract path
267                        path_match = re.match(r"['\"]?(/[^'\":]+)['\"]?:", stripped)
268                        if path_match:
269                            current_path = path_match.group(1)
270                            result["paths"][current_path] = {}
271                    elif current_path and stripped.rstrip(':') in ["get", "post", "put", "delete", "patch"]:
272                        method = stripped.rstrip(':')
273                        result["paths"][current_path][method] = {}
274
275                    j += 1
276
277        return result

280        """Check if OpenAPI method spec requires authentication."""
281        if "security" in method_spec:

285    def _discover_from_python(self, file_path: Path) -> list[QAEndpoint]:
286        """Discover endpoints from Python file (FastAPI, Flask, etc.)."""
287        endpoints: list[QAEndpoint] = []
288
289        try:
290            content = file_path.read_text()
291
292            # FastAPI patterns
293            # @router.get("/path") or @app.get("/path")
294            fastapi_patterns = [
295                (r'@(?:router|app)\.get\s*\(\s*["\']([^"\']+)["\']', "GET"),
296                (r'@(?:router|app)\.post\s*\(\s*["\']([^"\']+)["\']', "POST"),
297                (r'@(?:router|app)\.put\s*\(\s*["\']([^"\']+)["\']', "PUT"),
298                (r'@(?:router|app)\.delete\s*\(\s*["\']([^"\']+)["\']', "DELETE"),
299                (r'@(?:router|app)\.patch\s*\(\s*["\']([^"\']+)["\']', "PATCH"),
300            ]
301
302            for pattern, method in fastapi_patterns:
303                for match in re.finditer(pattern, content):
304                    path = match.group(1)
305                    # Check if auth is required (look for Depends in the line)
306                    line_start = content.rfind('\n', 0, match.start()) + 1
307                    line_end = content.find('\n', match.end())
308                    line = content[line_start:line_end if line_end > 0 else len(content)]
309                    auth_required = "Depends" in line or "get_current_user" in line
310
311                    endpoints.append(QAEndpoint(
312                        method=method,
313                        path=path,
314                        auth_required=auth_required,
315                    ))
316
317            # Flask patterns
318            # @app.route("/path", methods=["GET"])
319            flask_pattern = r'@(?:app|blueprint)\.route\s*\(\s*["\']([^"\']+)["\'](?:.*?methods\s*=\s*\[([^\]]+)\])?'
320            for match in re.finditer(flask_pattern, content, re.DOTALL):
321                path = match.group(1)
322                methods_str = match.group(2)
323
324                if methods_str:
325                    methods = re.findall(r'["\'](\w+)["\']', methods_str)
326                else:
327                    methods = ["GET"]
328
329                for method in methods:
330                    endpoints.append(QAEndpoint(
331                        method=method.upper(),
332                        path=path,
333                    ))
334
335        except Exception as e:
336            self._log("python_parse_error", {"error": str(e)}, level="warning")
337
338        return endpoints

410    def find_consumers(self, endpoints: list[QAEndpoint]) -> dict[str, list[str]]:
411        """
412        Find code that calls each endpoint.
413
414        Searches for:
415        1. Frontend fetch/axios calls
416        2. Other service HTTP calls
417        3. Integration tests
418
419        Args:
420            endpoints: List of endpoints to find consumers for.
421
422        Returns:
423            Dictionary mapping endpoint paths to list of consumer file paths.
424        """
425        consumers: dict[str, list[str]] = {}
426
427        for endpoint in endpoints:
428            consumers[endpoint.path] = []
429
430        # Search patterns
431        search_patterns = [
432            # JavaScript/TypeScript fetch
433            r'fetch\s*\(\s*["\'][^"\']*({path})',
434            # Axios
435            r'axios\.(get|post|put|delete)\s*\(\s*["\'][^"\']*({path})',
436            # Python requests
437            r'requests\.(get|post|put|delete)\s*\([^)]*({path})',
438            # httpx
439            r'client\.(get|post|put|delete)\s*\([^)]*({path})',
440        ]
441
442        # Search in common directories
443        search_dirs = [
444            self.repo_root / "frontend",
445            self.repo_root / "src",
446            self.repo_root / "clients",
447            self.repo_root / "tests",
448        ]
449
450        for search_dir in search_dirs:
451            if not search_dir.exists():
452                continue
453
454            for ext in ["*.ts", "*.tsx", "*.js", "*.jsx", "*.py"]:
455                for file_path in search_dir.rglob(ext):
456                    try:
457                        content = file_path.read_text()
458
459                        for endpoint in endpoints:
460                            # Escape special regex chars in path
461                            escaped_path = re.escape(endpoint.path)
462                            # Allow for path parameters
463                            pattern_path = escaped_path.replace(r'\{', r'[^/]*').replace(r'\}', '')
464
465                            for pattern in search_patterns:
466                                full_pattern = pattern.format(path=pattern_path)
467                                if re.search(full_pattern, content):
468                                    consumers[endpoint.path].append(str(file_path))
469                                    break
470
471                            # Also check for literal path in file
472                            if endpoint.path in content:
473                                if str(file_path) not in consumers[endpoint.path]:
474                                    consumers[endpoint.path].append(str(file_path))
475
476                    except Exception:
477                        pass
478
479        return consumers

416        2. Other service HTTP calls
417        3. Integration tests
418
419        Args:
420            endpoints: List of endpoints to find consumers for.
421
422        Returns:
423            Dictionary mapping endpoint paths to list of consumer file paths.
424        """
425        consumers: dict[str, list[str]] = {}
426
427        for endpoint in endpoints:
428            consumers[endpoint.path] = []
429
430        # Search patterns
431        search_patterns = [
432            # JavaScript/TypeScript fetch
433            r'fetch\s*\(\s*["\'][^"\']*({path})',
434            # Axios
435            r'axios\.(get|post|put|delete)\s*\(\s*["\'][^"\']*({path})',
436            # Python requests
437            r'requests\.(get|post|put|delete)\s*\([^)]*({path})',
438            # httpx
439            r'client\.(get|post|put|delete)\s*\([^)]*({path})',
440        ]
441
442        # Search in common directories
443        search_dirs = [
444            self.repo_root / "frontend",
445            self.repo_root / "src",
446            self.repo_root / "clients",
447            self.repo_root / "tests",
448        ]
449
450        for search_dir in search_dirs:
451            if not search_dir.exists():
452                continue
453
454            for ext in ["*.ts", "*.tsx", "*.js", "*.jsx", "*.py"]:
455                for file_path in search_dir.rglob(ext):
456                    try:
457                        content = file_path.read_text()
458
459                        for endpoint in endpoints:
460                            # Escape special regex chars in path
461                            escaped_path = re.escape(endpoint.path)
462                            # Allow for path parameters
463                            pattern_path = escaped_path.replace(r'\{', r'[^/]*').replace(r'\}', '')
464
465                            for pattern in search_patterns:
466                                full_pattern = pattern.format(path=pattern_path)
467                                if re.search(full_pattern, content):
468                                    consumers[endpoint.path].append(str(file_path))
469                                    break
470
471                            # Also check for literal path in file
472                            if endpoint.path in content:
473                                if str(file_path) not in consumers[endpoint.path]:
474                                    consumers[endpoint.path].append(str(file_path))
475
476                    except Exception:
477                        pass
478
479        return consumers
