 96    def run_post_verification_qa(
 97        self,
 98        feature_id: str,
 99        issue_number: int,
100        verified_files: list[str],
101        skip_qa: bool = False,
102    ) -> QAIntegrationResult:
103        """
104        Run QA after successful verification.
105
106        Args:
107            feature_id: The feature identifier.
108            issue_number: The issue number.
109            verified_files: List of file paths that were verified.
110            skip_qa: If True, skip QA and return PASS immediately.
111
112        Returns:
113            QAIntegrationResult with session details and recommendations.
114        """
115        # Skip QA if requested
116        if skip_qa:
117            self._log("qa_skipped", {"feature_id": feature_id, "issue_number": issue_number})
118            return QAIntegrationResult(
119                session_id=None,
120                recommendation=QARecommendation.PASS,
121                should_block=False,
122            )
123
124        # Determine depth based on file risk
125        depth = QADepth.STANDARD
126        if self._has_high_risk_files(verified_files):
127            depth = QADepth.DEEP
128            self._log("qa_depth_escalated", {
129                "feature_id": feature_id,
130                "reason": "high_risk_files_detected",
131            })
132
133        try:
134            # Run QA validation
135            session = self.orchestrator.validate_issue(
136                feature_id=feature_id,
137                issue_number=issue_number,
138                depth=depth,
139            )
140
141            # Build result
142            result = QAIntegrationResult(
143                session_id=session.session_id,
144            )
145
146            if session.result:
147                result.recommendation = session.result.recommendation
148
149                # Check if should block
150                should_block, block_reason = self.should_block_commit(session)
151                result.should_block = should_block
152                result.block_reason = block_reason
153
154                # Build findings summary
155                result.findings_summary = {
156                    "critical": session.result.critical_count,
157                    "moderate": session.result.moderate_count,
158                    "minor": session.result.minor_count,
159                }
160
161            self._log("qa_post_verification_complete", {
162                "session_id": session.session_id,
163                "recommendation": result.recommendation.value,
164                "should_block": result.should_block,
165            })
166
167            return result
168
169        except Exception as e:
170            self._log("qa_post_verification_error", {"error": str(e)}, level="error")
171            return QAIntegrationResult(
172                error=str(e),
173                recommendation=QARecommendation.PASS,  # Fail open
174                should_block=False,
175            )

177    def should_block_commit(
178        self,
179        qa_result: QASession,
180    ) -> tuple[bool, Optional[str]]:
181        """
182        Determine if QA findings should block commit.
183
184        Args:
185            qa_result: The QASession with results.
186
187        Returns:
188            Tuple of (should_block, reason). Reason is None if not blocking.
189        """
190        # No result means session failed - fail open
191        if qa_result.result is None:
192            return False, None
193
194        # Check recommendation
195        if qa_result.result.recommendation == QARecommendation.BLOCK:
196            critical_count = qa_result.result.critical_count
197            finding_details = ""
198            if qa_result.result.findings:
199                first_finding = qa_result.result.findings[0]
200                finding_details = f": {first_finding.title}"
201
202            reason = f"QA found {critical_count} critical issue(s){finding_details}"
203            return True, reason
204
205        # WARN and PASS do not block
206        return False, None

 75    def enhance_reproduction(
 76        self,
 77        bug_id: str,
 78        bug_description: str,
 79        error_message: Optional[str] = None,
 80        affected_endpoints: Optional[list[str]] = None,
 81    ) -> BugReproductionResult:
 82        """
 83        Attempt behavioral reproduction of a bug.
 84
 85        Uses DEEP depth to thoroughly test the affected area with edge cases,
 86        error cases, and security probes to find the reproduction path.
 87
 88        Args:
 89            bug_id: The bug identifier.
 90            bug_description: Description of the bug from the report.
 91            error_message: Optional error message from the bug report.
 92            affected_endpoints: Optional list of endpoints to focus on.
 93
 94        Returns:
 95            BugReproductionResult with reproduction status and evidence.
 96        """
 97        # Build target string from available info
 98        target = self._build_target(bug_description, error_message, affected_endpoints)
 99
100        self._log("qa_bug_reproduction_start", {
101            "bug_id": bug_id,
102            "target": target,
103        })
104
105        try:
106            # Run QA with DEEP depth for thorough bug hunting
107            session = self.orchestrator.test(
108                target=target,
109                depth=QADepth.DEEP,
110                trigger=QATrigger.BUG_REPRODUCTION,
111            )
112
113            # Build result
114            result = BugReproductionResult(
115                session_id=session.session_id,
116            )
117
118            if session.result:
119                # Bug is reproduced if QA found any failures
120                result.is_reproduced = session.result.tests_failed > 0
121
122                # Extract reproduction steps from findings
123                if session.result.findings:
124                    result.reproduction_steps = self._extract_reproduction_steps(
125                        session.result.findings
126                    )
127                    result.evidence = self._extract_evidence(session.result.findings)
128                    result.root_cause_hints = self._extract_root_cause_hints(
129                        session.result.findings
130                    )
131
132            self._log("qa_bug_reproduction_complete", {
133                "bug_id": bug_id,
134                "session_id": session.session_id,
135                "is_reproduced": result.is_reproduced,
136            })
137
138            return result
139
140        except Exception as e:
141            self._log("qa_bug_reproduction_error", {"error": str(e)}, level="error")
142            return BugReproductionResult(
143                error=str(e),
144                is_reproduced=False,
145            )

147    def get_rca_evidence(
148        self,
149        session_id: str,
150    ) -> dict[str, Any]:
151        """
152        Extract evidence for root cause analysis.
153
154        Gathers all relevant evidence from a QA session for consumption
155        by the RootCauseAnalyzer agent.
156
157        Args:
158            session_id: The QA session ID.
159
160        Returns:
161            Dictionary with evidence formatted for RCA:
162            - endpoints_affected: List of affected endpoints
163            - findings: List of finding summaries
164            - requests: List of request evidence
165            - responses: List of response evidence
166            - stack_traces: List of stack traces if available
167            - reproduction_steps: List of steps to reproduce
168        """
169        session = self.orchestrator.get_session(session_id)
170        if not session:
171            return {}
172
173        if not session.result:
174            return {"session_id": session_id}
175
176        evidence: dict[str, Any] = {
177            "session_id": session_id,
178            "endpoints_affected": [],
179            "findings": [],
180            "requests": [],
181            "responses": [],
182            "stack_traces": [],
183            "reproduction_steps": [],
184        }
185
186        seen_endpoints: set[str] = set()
187
188        for finding in session.result.findings:
189            # Track unique endpoints
190            if finding.endpoint not in seen_endpoints:
191                evidence["endpoints_affected"].append(finding.endpoint)
192                seen_endpoints.add(finding.endpoint)
193
194            # Add finding summary
195            evidence["findings"].append({
196                "id": finding.finding_id,
197                "severity": finding.severity,
198                "title": finding.title,
199                "endpoint": finding.endpoint,
200                "expected": finding.expected,
201                "actual": finding.actual,
202            })
203
204            # Extract request/response evidence
205            if "request" in finding.evidence:
206                evidence["requests"].append(finding.evidence["request"])
207            if "response" in finding.evidence:
208                evidence["responses"].append(finding.evidence["response"])
209            if "stack_trace" in finding.evidence:
210                evidence["stack_traces"].append(finding.evidence["stack_trace"])
211
212        # Generate reproduction steps
213        evidence["reproduction_steps"] = self._extract_reproduction_steps(
214            session.result.findings
215        )
216
217        return evidence

219    def _build_target(
220        self,
221        bug_description: str,
222        error_message: Optional[str],
223        affected_endpoints: Optional[list[str]],
224    ) -> str:
225        """Build a target string for QA testing."""
226        if affected_endpoints:
227            # Use first endpoint as primary target
228            return affected_endpoints[0]
229
230        # Extract potential endpoint from description
231        description_lower = bug_description.lower()
232
233        # Look for common API patterns
234        if "/api/" in description_lower:
235            # Extract endpoint pattern
236            start = description_lower.find("/api/")
237            end = description_lower.find(" ", start)
238            if end == -1:
239                end = len(description_lower)
240            return description_lower[start:end]
241
242        # Use description as-is
243        return bug_description

245    def _extract_reproduction_steps(
246        self,
247        findings: list[QAFinding],
248    ) -> list[str]:
249        """Extract reproduction steps from findings."""
250        steps: list[str] = []
251
252        for finding in findings:
253            # Build step from evidence
254            step_parts = []
255
256            if "request" in finding.evidence:
257                step_parts.append(f"Execute: {finding.evidence['request']}")
258            else:
259                step_parts.append(f"Send request to {finding.endpoint}")
260
261            step_parts.append(f"Expected: {finding.expected}")
262            step_parts.append(f"Actual: {finding.actual}")
263
264            if steps:
265                steps.append("---")  # Separator between findings
266            steps.extend(step_parts)
267
268        return steps

270    def _extract_evidence(
271        self,
272        findings: list[QAFinding],
273    ) -> dict[str, Any]:
274        """Extract all evidence from findings."""
275        evidence: dict[str, Any] = {
276            "requests": [],
277            "responses": [],
278            "endpoints": [],
279        }
280
281        for finding in findings:
282            if "request" in finding.evidence:
283                evidence["requests"].append(finding.evidence["request"])
284            if "response" in finding.evidence:
285                evidence["responses"].append(finding.evidence["response"])
286            evidence["endpoints"].append(finding.endpoint)
287
288        # Deduplicate endpoints
289        evidence["endpoints"] = list(set(evidence["endpoints"]))
290
291        return evidence

293    def _extract_root_cause_hints(
294        self,
295        findings: list[QAFinding],
296    ) -> list[str]:
297        """Extract potential root cause hints from findings."""
298        hints: list[str] = []
