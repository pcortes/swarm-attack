136    def start_service(self, base_url: str) -> ServiceStartupResult:
137        """
138        Start the service and wait for it to be healthy.
139
140        Args:
141            base_url: Base URL where service will be available.
142
143        Returns:
144            ServiceStartupResult indicating success or failure reason.
145        """
146        parsed = urlparse(base_url)
147        port = parsed.port or 8000
148        host = parsed.hostname or "127.0.0.1"
149
150        # Check port availability first (Section 10.1)
151        if not check_port_available(port, host):
152            self._log("port_conflict", {"port": port}, level="error")
153            return ServiceStartupResult.PORT_CONFLICT
154
155        # Try to start the service
156        if not self._start_service_process():
157            return ServiceStartupResult.STARTUP_CRASHED
158
159        # Wait for health endpoint with timeout
160        start_time = time.time()
161        health_endpoint = None
162
163        while time.time() - start_time < self.startup_timeout_seconds:
164            health_endpoint = self._try_health_endpoints(base_url)
165            if health_endpoint:
166                # Verify stability (Section 10.1)
167                if self._verify_service_stability(base_url):
168                    self._log("service_started", {
169                        "health_endpoint": health_endpoint,
170                        "startup_time": time.time() - start_time,
171                    })
172                    return ServiceStartupResult.SUCCESS
173            time.sleep(0.5)
174
175        # Timeout reached
176        if health_endpoint is None:
177            return ServiceStartupResult.NO_HEALTH_ENDPOINT
178        return ServiceStartupResult.TIMEOUT

180    def _start_service_process(self) -> bool:
181        """
182        Attempt to start the service process.
183
184        Looks for common startup methods:
185        - docker-compose.yml
186        - Makefile with dev target
187        - scripts/dev.sh
188
189        Returns:
190            True if process started, False otherwise.
191        """
192        repo_root = Path(self.config.repo_root)
193
194        # Try docker-compose first
195        docker_compose = repo_root / "docker-compose.yml"
196        if docker_compose.exists():
197            try:
198                self._service_process = subprocess.Popen(
199                    ["docker-compose", "up", "-d"],
200                    cwd=repo_root,
201                    stdout=subprocess.PIPE,
202                    stderr=subprocess.PIPE,
203                )
204                return True
205            except Exception as e:
206                self._log("docker_start_failed", {"error": str(e)}, level="error")
207
208        # Try Makefile
209        makefile = repo_root / "Makefile"
210        if makefile.exists():
211            try:
212                self._service_process = subprocess.Popen(
213                    ["make", "dev"],
214                    cwd=repo_root,
215                    stdout=subprocess.PIPE,
216                    stderr=subprocess.PIPE,
217                )
218                return True
219            except Exception:
220                pass
221
222        # Try scripts/dev.sh
223        dev_script = repo_root / "scripts" / "dev.sh"
224        if dev_script.exists():
225            try:
226                self._service_process = subprocess.Popen(
227                    ["bash", str(dev_script)],
228                    cwd=repo_root,
229                    stdout=subprocess.PIPE,
230                    stderr=subprocess.PIPE,
231                )
232                return True
233            except Exception:
234                pass
235
236        # No startup method found - assume service is already running
237        return True

239    def _try_health_endpoints(self, base_url: str) -> Optional[str]:
240        """
241        Try health endpoints in order until one succeeds.
242
243        Args:
244            base_url: Base URL for the service.
245
246        Returns:
247            The path of the first healthy endpoint, or None if all fail.
248        """
249        for endpoint in self.health_endpoints:
250            url = f"{base_url.rstrip('/')}{endpoint}"
251            try:
252                response = requests.get(url, timeout=5)
253                # Accept any 2xx for dedicated health endpoints
254                if endpoint != "/" and 200 <= response.status_code < 300:
255                    return endpoint
256                # For root path, accept any response (means server is running)
257                if endpoint == "/":
258                    return endpoint
259            except Exception:
260                continue
261        return None

263    def _verify_service_stability(self, base_url: str) -> bool:
264        """
265        Verify service remains healthy after initial startup.
266
267        Waits 2-3 seconds and checks health again to catch
268        services that crash immediately after startup (Section 10.1).
269
270        Args:
271            base_url: Base URL for the service.
272
273        Returns:
274            True if service is stable, False if it crashed.
275        """
276        # Initial health check
277        initial = self._try_health_endpoints(base_url)
278        if not initial:
279            return False
280
281        # Wait for stability
282        time.sleep(self.stability_check_delay_seconds)
283
284        # Check again
285        final = self._try_health_endpoints(base_url)
286        return final is not None

420        self,
421        endpoint: QAEndpoint,
422        test_case: dict[str, Any],
423        base_url: str,
424        resilience: ResilienceConfig,
425    ) -> dict[str, Any]:
426        """
427        Execute a single test case.
428
429        Args:
430            endpoint: The endpoint being tested.
431            test_case: Test case specification.
432            base_url: Base URL for requests.
433            resilience: Retry configuration.
434
435        Returns:
436            Dictionary with test result and optional finding.
437        """
438        url = f"{base_url.rstrip('/')}{endpoint.path}"
439        method = endpoint.method
440        body = test_case.get("body")
441        expected_status = test_case.get("expected_status", 200)
442
443        try:
444            response = self._make_request_with_retry(
445                method, url, json=body, resilience=resilience
446            )
447
448            passed = response.status_code == expected_status
449
450            if not passed:
451                finding = self._create_finding(
452                    endpoint=f"{method} {endpoint.path}",
453                    test_type=test_case["test_type"],
454                    expected={"status": expected_status},
455                    actual={"status": response.status_code},
456                    request_evidence=f"curl -X {method} {url}",
457                    response_evidence=response.text[:500] if response.text else "",
458                )
459                return {"passed": False, "finding": finding}
460
461            return {"passed": True}
462
463        except Exception as e:
464            finding = self._create_finding(
465                endpoint=f"{method} {endpoint.path}",
466                test_type=test_case["test_type"],
467                expected={"status": expected_status},
468                actual={"error": str(e)},
469                request_evidence=f"curl -X {method} {url}",
470                response_evidence=str(e),
471            )
472            return {"passed": False, "finding": finding}

474    def _make_request_with_retry(
475        self,
476        method: str,
477        url: str,
478        resilience: ResilienceConfig,
479        **kwargs: Any,
480    ) -> requests.Response:
481        """
482        Make an HTTP request with retry and backoff logic.
483
484        Args:
485            method: HTTP method.
486            url: Request URL.
487            resilience: Retry configuration.
488            **kwargs: Additional arguments for requests.
489
490        Returns:
491            Response object.
492
493        Raises:
494            Exception if all retries exhausted.
495        """
496        last_exception: Optional[Exception] = None
497        delay = resilience.retry_backoff_seconds
498
499        for attempt in range(resilience.max_retries):
500            try:
501                response = requests.request(
502                    method,
503                    url,
504                    timeout=resilience.request_timeout_seconds,
505                    **kwargs,
506                )
507
508                # Check for rate limiting
509                is_rate_limited, _ = self._check_rate_limit(response, resilience)
510                if is_rate_limited and attempt < resilience.max_retries - 1:
511                    retry_after = self._get_retry_after(response, resilience)
512                    time.sleep(retry_after)
513                    continue
514
515                # Check for retryable status codes
516                if response.status_code in resilience.retry_on_status:
517                    if attempt < resilience.max_retries - 1:
518                        time.sleep(delay)
519                        delay *= 2  # Exponential backoff
520                        continue
521
522                return response
523
524            except (ConnectionError, requests.exceptions.RequestException) as e:
525                last_exception = e
526                if attempt < resilience.max_retries - 1:
527                    time.sleep(delay)
528                    delay *= 2
529
530        if last_exception:
531            raise last_exception
532        raise Exception("Request failed after all retries")

569    def _create_finding(
570        self,
571        endpoint: str,
572        test_type: str,
573        expected: dict[str, Any],
574        actual: dict[str, Any],
575        request_evidence: str,
576        response_evidence: str,
577    ) -> QAFinding:
578        """
579        Create a QAFinding from test failure.
580
581        Args:
582            endpoint: Endpoint being tested.
583            test_type: Type of test that failed.
584            expected: Expected values.
585            actual: Actual values.
586            request_evidence: Request details.
587            response_evidence: Response details.
588
589        Returns:
590            QAFinding object.
591        """
592        self._finding_counter += 1
593        finding_id = f"BT-{self._finding_counter:03d}"
594
595        # Determine severity
596        actual_status = actual.get("status", 0)
597        if actual_status >= 500:
598            severity = "critical"
599            title = f"Server error ({actual_status})"
600        elif actual.get("error"):
601            severity = "critical"
602            title = "Request failed"
603        elif expected.get("status", 200) != actual_status:
604            severity = "moderate"
605            title = f"Wrong status code (expected {expected.get('status')}, got {actual_status})"
606        else:
607            severity = "minor"
608            title = "Unexpected response"
609
610        return QAFinding(
611            finding_id=finding_id,
612            severity=severity,
613            category="behavioral",
614            endpoint=endpoint,
615            test_type=test_type,
616            title=title,
617            description=f"Test '{test_type}' failed for endpoint {endpoint}",
618            expected=expected,
619            actual=actual,
620            evidence={
621                "request": request_evidence,
622                "response": response_evidence,
623            },
624            recommendation=self._get_recommendation(severity, test_type),
625        )
