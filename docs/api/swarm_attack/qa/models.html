256@dataclass
257class QASession:
258    """Complete state of a QA session."""
259    session_id: str
260    trigger: QATrigger
261    depth: QADepth
262    status: QAStatus
263    context: QAContext
264    result: Optional[QAResult] = None
265    created_at: str = ""
266    started_at: Optional[str] = None
267    completed_at: Optional[str] = None
268    cost_usd: float = 0.0
269    error: Optional[str] = None
270
271    def __post_init__(self) -> None:
272        if not self.created_at:
273            self.created_at = datetime.now(timezone.utc).isoformat().replace("+00:00", "Z")
274
275    def to_dict(self) -> dict[str, Any]:
276        return {
277            "session_id": self.session_id,
278            "trigger": self.trigger.value,
279            "depth": self.depth.value,
280            "status": self.status.value,
281            "context": self.context.to_dict(),
282            "result": self.result.to_dict() if self.result else None,
283            "created_at": self.created_at,
284            "started_at": self.started_at,
285            "completed_at": self.completed_at,
286            "cost_usd": self.cost_usd,
287            "error": self.error,
288        }
289
290    @classmethod
291    def from_dict(cls, data: dict[str, Any]) -> QASession:
292        trigger = QATrigger(data["trigger"])
293        depth = QADepth(data["depth"])
294        status = QAStatus(data["status"])
295        context_data = data.get("context", {})
296        context = QAContext(
297            feature_id=context_data.get("feature_id"),
298            issue_number=context_data.get("issue_number"),
299            bug_id=context_data.get("bug_id"),
300            target_files=context_data.get("target_files", []),
301        )
302        result = None
303        if data.get("result"):
304            result_data = data["result"]
305            findings = [QAFinding.from_dict(f) for f in result_data.get("findings", [])]
306            result = QAResult(
307                tests_run=result_data.get("tests_run", 0),
308                tests_passed=result_data.get("tests_passed", 0),
309                tests_failed=result_data.get("tests_failed", 0),
310                findings=findings,
311                recommendation=QARecommendation(result_data.get("recommendation", "pass")),
312            )
313        return cls(
314            session_id=data["session_id"],
315            trigger=trigger,
316            depth=depth,
317            status=status,
318            context=context,
319            result=result,
320            created_at=data.get("created_at", ""),
321            started_at=data.get("started_at"),
322            completed_at=data.get("completed_at"),
323            cost_usd=data.get("cost_usd", 0.0),
324            error=data.get("error"),
325        )
